
## Convert fasta samples to fastq

for j in {001..016}; 
do  
# Read 1
python fasta2fastq.py -o linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/sample_01_1.fq linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/sample_01_1.fasta; 

# Read 2
python fasta2fastq.py -o linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/sample_01_2.fq linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/sample_01_2.fasta; 
done


for j in {001..016}; 
do 
# Read 1
last_header=$(grep "^@" circRNA_simulated_reads/circ_reads_chrAll_coverage2_error1/Sim_"$j"_circ_1.fq | tail -n 1 | cut -d ' ' -f 1); start_num=$(echo $last_header | cut -d '.' -f 2); awk -v start="$start_num" -v j="$j" 'BEGIN {i=start+1} /^@/ {$0 = "@simulate:SRR"j"."i" length=101"; i++} {print}' linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/sample_01_1.fq > linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/header_sample_01_1.fq; 

# Read 2
last_header=$(grep "^@" circRNA_simulated_reads/circ_reads_chrAll_coverage2_error1/Sim_"$j"_circ_2.fq | tail -n 1 | cut -d ' ' -f 1); start_num=$(echo $last_header | cut -d '.' -f 2); awk -v start="$start_num" -v j="$j" 'BEGIN {i=start+1} /^@/ {$0 = "@simulate:SRR"j"."i" length=101"; i++} {print}' linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/sample_01_2.fq > linearRNA_simulated_reads/lin_reads_chrAll_coverage100_error1/SIM_"$j"/header_sample_01_2.fq; 
done
